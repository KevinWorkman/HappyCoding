function setup() {
  createCanvas(500, 500);
}

function draw() {

  // gray background
  background(100);

  // stem
  stroke(0, 160, 0);
  strokeWeight(20);
  line(250, 150, 150, 100);

  // orange
  fill(255, 100, 0);
  stroke(0, 100);
  strokeWeight(3);

  // pumpkin is made up of circles
  ellipse(250, 250, 400, 200);
  ellipse(250, 250, 300, 200);
  ellipse(250, 250, 200, 200);
  ellipse(250, 250, 100, 200);
}
</script>
